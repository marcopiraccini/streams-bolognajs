Zipping a file: the stream way


var fs = require('fs');
var zlib = require('zlib');
var file = process.argv[2];

fs.createReadStream(file)
    .pipe(zlib.createGzip())
    .pipe(fs.createWriteStream(file + '.stream.gz'))
        .on('finish', function() {
            console.log('File successfully compressed');
    });
