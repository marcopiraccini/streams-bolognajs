  • Subclass di stream.Writable (o estensione un Writeable object)
  • Implementa _write(chunk, encoding, callback).

var Writable = require('stream').Writable;
var ws = Writable();
ws._write = function (chunk, enc, next) {
    console.log(chunk);
    next();
};

process.stdin.pipe(ws);
