  • Gli Stream Readeable sono implementati subclassando stream.Readable (o estendendo un Readeable object) e implementando_read(size).
  • size in bytes, ma pu√≤ esssere ignorato (objectMode)
  • _read(size) deve chiamare this.push(chunk) per mandare un chunk al consumer

var Readable = require('stream').Readable;
var rs = Readable();

var c = 97;
rs._read = function () {
    rs.push(String.fromCharCode(c++));
    if (c > 'z'.charCodeAt(0)) rs.push(null);
};

rs.pipe(process.stdout);
